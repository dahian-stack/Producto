CREATE DATABASE mediapro;

USE mediapro;

SHOW TABLES;

CREATE TABLE Cliente (
    ID_cliente INTEGER PRIMARY KEY AUTO_INCREMENT,
    Nombre VARCHAR(100),
    Tipo VARCHAR(50),
    Contacto VARCHAR(100)
);

SELECT * FROM cliente;

CREATE TABLE Produccion (
    ID_produccion INT PRIMARY KEY AUTO_INCREMENT,
    Titulo VARCHAR(150),
    Tipo VARCHAR(50),
    Genero VARCHAR(50),
    Presupuesto_inicial DECIMAL(15,2),
    Presupuesto_gastado DECIMAL(15,2),
    Estado_presupuesto VARCHAR(50),
    Fecha_inicio DATE,
    Duracion_estimada INT,
    Etapa VARCHAR(50),
    Estado VARCHAR(50),
    ID_cliente INT,
    FOREIGN KEY (ID_cliente) REFERENCES Cliente(ID_cliente)
);

SELECT * FROM Produccion;

CREATE TABLE Personal (
    ID_personal INTEGER PRIMARY KEY AUTO_INCREMENT,
    Nombres VARCHAR(100),
    Apellidos VARCHAR(100),
    Especialidad VARCHAR(100),
    Experiencia TEXT,
    Portafolio TEXT,
    Disponibilidad VARCHAR(50),
    Tarifa DECIMAL(10,2),
    Contacto VARCHAR(100)
);

SELECT * FROM Personal;

CREATE TABLE Produccion_Personal (
    ID_produccion INTEGER,
    ID_personal INTEGER,
    Rol VARCHAR(50),
    FOREIGN KEY (ID_produccion) REFERENCES Produccion(ID_produccion),
    FOREIGN KEY (ID_personal) REFERENCES Personal(ID_personal)
);

SELECT * FROM Produccion_Personal;

CREATE TABLE Equipo_Tecnico (
    ID_equipo INT PRIMARY KEY AUTO_INCREMENT,
    Categoria VARCHAR(50),
    Marca VARCHAR(50),
    Modelo VARCHAR(50),
    Caracteristicas TEXT,
    Fecha_adquisicion DATE,
    Valor DECIMAL(12,2),
    Estado VARCHAR(50),
    Disponibilidad VARCHAR(50),
    Seguro VARCHAR(50),
    Proximo_mantenimiento DATE,
    Estado_mantenimiento VARCHAR(50)
);

SELECT * FROM Equipo_Tecnico;

CREATE TABLE Mantenimiento (
    ID_mantenimiento INTEGER PRIMARY KEY AUTO_INCREMENT,
    ID_equipo INTEGER,
    Fecha DATE,
    Tipo VARCHAR(50),
    Descripcion TEXT,
    FOREIGN KEY (ID_equipo) REFERENCES Equipo_Tecnico(ID_equipo)
);

SELECT * FROM Mantenimiento;

CREATE TABLE Produccion_Equipo (
    ID_produccion INTEGER,
    ID_equipo INTEGER,
    FOREIGN KEY (ID_produccion) REFERENCES Produccion(ID_produccion),
    FOREIGN KEY (ID_equipo) REFERENCES Equipo_Tecnico(ID_equipo)
);

SELECT * FROM Produccion_Equipo;

CREATE TABLE Localizacion (
    ID_localizacion INT PRIMARY KEY AUTO_INCREMENT,
    Nombre VARCHAR(100),
    Direccion VARCHAR(150),
    Coordenadas VARCHAR(100),
    Tipo VARCHAR(50),
    Propietario VARCHAR(100),
    Caracteristicas TEXT,
    Permisos TEXT,
    Estado VARCHAR(50),
    Costo DECIMAL(10,2),
    Limitaciones TEXT,
    Facilidades TEXT,
    Accesibilidad TEXT
);

SELECT * FROM Localizacion;

CREATE TABLE Produccion_Localizacion (
    ID_produccion INTEGER,
    ID_localizacion INTEGER,
    FOREIGN KEY (ID_produccion) REFERENCES Produccion(ID_produccion),
    FOREIGN KEY (ID_localizacion) REFERENCES Localizacion(ID_localizacion)
);

SELECT * FROM Produccion_Localizacion;

CREATE TABLE Plan_Rodaje (
    ID_plan INT PRIMARY KEY AUTO_INCREMENT,
    ID_produccion INT,
    Dia_rodaje INT,
    Fecha DATE,
    Hora_inicio TIME,
    Hora_fin TIME,
    Hora_inicio_real TIME,
    Hora_fin_real TIME,
    Eficiencia VARCHAR(50),
    Requerimientos TEXT,
    Plan_contingencia TEXT,
    ID_localizacion INT,
    FOREIGN KEY (ID_produccion) REFERENCES Produccion(ID_produccion),
    FOREIGN KEY (ID_localizacion) REFERENCES Localizacion(ID_localizacion)
);

SELECT * FROM Plan_Rodaje;

CREATE TABLE Escena (
    ID_escena INTEGER AUTO_INCREMENT PRIMARY KEY,
    ID_produccion INT,
    Descripcion TEXT,
    Tiempo_estimado INTEGER,
    Complejidad VARCHAR(50),
    Necesidades_especiales TEXT,
    Estado VARCHAR(50),
    FOREIGN KEY (ID_produccion) REFERENCES Produccion(ID_produccion)
);

SELECT * FROM Escena;

CREATE TABLE Plan_Rodaje_Escena (
    ID_plan INTEGER,
    ID_escena INTEGER,
    FOREIGN KEY (ID_plan) REFERENCES Plan_Rodaje(ID_plan),
    FOREIGN KEY (ID_escena) REFERENCES Escena(ID_escena)
);

SELECT * FROM Plan_Rodaje_Escena;

CREATE TABLE Actor (
    ID_actor INT PRIMARY KEY AUTO_INCREMENT,
    Nombres VARCHAR(100),
    Apellidos VARCHAR(100),
    Documento VARCHAR(20),
    Fecha_nacimiento DATE,
    Representante VARCHAR(100),
    Caracteristicas_fisicas TEXT,
    Habilidades TEXT,
    Trabajos_previos TEXT,
    Disponibilidad VARCHAR(50),
    Categoria VARCHAR(50),
    Salario DECIMAL(10,2)
);

SELECT * FROM actor;

CREATE TABLE Personaje (
    ID_personaje INTEGER PRIMARY KEY AUTO_INCREMENT,
    Nombre VARCHAR(100),
    Descripcion TEXT
);

SELECT * FROM Personaje;

CREATE TABLE Escena_Actor (
    ID_escena INTEGER,
    ID_actor INTEGER,
    ID_personaje INTEGER,
    FOREIGN KEY (ID_escena) REFERENCES Escena(ID_escena),
    FOREIGN KEY (ID_actor) REFERENCES Actor(ID_actor),
    FOREIGN KEY (ID_personaje) REFERENCES Personaje(ID_personaje)
);

SELECT * FROM Escena_Actor;

CREATE TABLE Plan_Rodaje_Actor (
    ID_plan INTEGER,
    ID_actor INTEGER,
    FOREIGN KEY (ID_plan) REFERENCES Plan_Rodaje(ID_plan),
    FOREIGN KEY (ID_actor) REFERENCES Actor(ID_actor)
);

SELECT * FROM Plan_Rodaje_Actor;

CREATE TABLE Plan_Rodaje_Equipo (
    ID_plan INTEGER,
    ID_equipo INTEGER,
    FOREIGN KEY (ID_plan) REFERENCES Plan_Rodaje(ID_plan),
    FOREIGN KEY (ID_equipo) REFERENCES Equipo_Tecnico(ID_equipo)
);

SELECT * FROM Plan_Rodaje_Equipo;

CREATE TABLE Material_Grabado (
    ID_material INTEGER PRIMARY KEY AUTO_INCREMENT,
    ID_produccion INTEGER,
    ID_escena INTEGER,
    Toma INTEGER,
    Duracion INTEGER,
    Formato VARCHAR(50),
    Resolucion VARCHAR(50),
    Camara VARCHAR(50),
    Almacenamiento VARCHAR(100),
    Timecode_inicio VARCHAR(50),
    Timecode_fin VARCHAR(50),
    Observaciones TEXT,
    Calificacion VARCHAR(50),
    FOREIGN KEY (ID_produccion) REFERENCES Produccion(ID_produccion),
    FOREIGN KEY (ID_escena) REFERENCES Escena(ID_escena)
);

SELECT * FROM Material_Grabado;

CREATE TABLE Audio (
    ID_audio INTEGER PRIMARY KEY AUTO_INCREMENT,
    ID_material INTEGER,
    Tipo VARCHAR(50),
    Descripcion TEXT,
    FOREIGN KEY (ID_material) REFERENCES Material_Grabado(ID_material)
);

SELECT * FROM Audio;

CREATE TABLE Postproduccion (
    ID_post INT PRIMARY KEY AUTO_INCREMENT,
    ID_produccion INT,
    Tipo VARCHAR(50),
    Responsable VARCHAR(100),
    Fecha_inicio DATE,
    Fecha_fin DATE,
    Porcentaje INT,
    Avance_general INT,
    Software VARCHAR(100),
    Estado VARCHAR(50),
    FOREIGN KEY (ID_produccion) REFERENCES Produccion(ID_produccion)
);

SELECT * FROM Postproduccion;

CREATE TABLE Material_Postproduccion (
    ID_material_post INT AUTO_INCREMENT PRIMARY KEY,
    ID_post INT,
    Descripcion TEXT,
    FOREIGN KEY (ID_post) REFERENCES Postproduccion(ID_post)
);

SELECT * FROM Material_Postproduccion;

CREATE TABLE Versiones (
    ID_version INTEGER PRIMARY KEY AUTO_INCREMENT,
    ID_post INTEGER,
    Numero_version VARCHAR(20),
    Fecha DATE,
    Descripcion TEXT,
    FOREIGN KEY (ID_post) REFERENCES Postproduccion(ID_post)

SELECT * FROM Versiones;

CREATE TABLE Aprobacion (
    ID_aprobacion INTEGER PRIMARY KEY AUTO_INCREMENT,

SELECT * FROM Aprobacion;

CREATE TABLE Distribucion (
    ID_distribucion INTEGER PRIMARY KEY AUTO_INCREMENT,
    ID_produccion INTEGER,

-- IMPLEMENTAR ELIMINACIÓN DE CASCADA --

-- PRODUCCION -> CLIENTE
ALTER TABLE Produccion DROP FOREIGN KEY Produccion_ibfk_1;
ALTER TABLE Produccion
ADD FOREIGN KEY (id_cliente)
REFERENCES Cliente(id_cliente)
ON DELETE CASCADE;

-- ESCENA -> PRODUCCION
ALTER TABLE Escena DROP FOREIGN KEY Escena_ibfk_1;
ALTER TABLE Escena
ADD FOREIGN KEY (id_produccion)
REFERENCES Produccion(id_produccion)
ON DELETE CASCADE;

-- MATERIAL_GRABADO -> PRODUCCION
ALTER TABLE Material_Grabado DROP FOREIGN KEY Material_Grabado_ibfk_1;
ALTER TABLE Material_Grabado
ADD FOREIGN KEY (id_produccion)
REFERENCES Produccion(id_produccion)
ON DELETE CASCADE;

-- MATERIAL_GRABADO -> ESCENA
ALTER TABLE Material_Grabado DROP FOREIGN KEY Material_Grabado_ibfk_2;
ALTER TABLE Material_Grabado
ADD FOREIGN KEY (id_escena)
REFERENCES Escena(id_escena)
ON DELETE CASCADE;

-- AUDIO -> MATERIAL
ALTER TABLE Audio DROP FOREIGN KEY Audio_ibfk_1;
ALTER TABLE Audio
ADD FOREIGN KEY (id_material)
REFERENCES Material_Grabado(id_material)
ON DELETE CASCADE;

-- POSTPRODUCCION -> PRODUCCION
ALTER TABLE Postproduccion DROP FOREIGN KEY Postproduccion_ibfk_1;
ALTER TABLE Postproduccion
ADD FOREIGN KEY (id_produccion)
REFERENCES Produccion(id_produccion)
ON DELETE CASCADE;

-- 1. FN_CalcularPresupuestoEscena: Calcula el presupuesto necesario para una escena específica.
DELIMITER //
CREATE FUNCTION FN_CalcularPresupuestoEscena(p_id_escena INT)
RETURNS DECIMAL(10,2)
DETERMINISTIC
BEGIN
    DECLARE v_tiempo INT;
    DECLARE v_complejidad VARCHAR(50);
    DECLARE v_costo_minuto DECIMAL(10,2);
    DECLARE v_presupuesto DECIMAL(10,2);

    SELECT tiempo_estimado, complejidad
    INTO v_tiempo, v_complejidad
    FROM Escena
    WHERE ID_escena = p_id_escena;

    IF v_complejidad = 'baja' THEN
        SET v_costo_minuto = 100;
    ELSEIF v_complejidad = 'media' THEN
        SET v_costo_minuto = 200;
    ELSE
        SET v_costo_minuto = 300;
    END IF;

    SET v_presupuesto = v_tiempo * v_costo_minuto;

    RETURN v_presupuesto;
END //

DELIMITER ;

-- Ver el calculo con un solo ID
SELECT FN_CalcularPresupuestoEscena(7); 

-- O con todas las escenas
SELECT 
    ID_escena,
    descripcion,
    FN_CalcularPresupuestoEscena(ID_escena) AS presupuesto
FROM escena;

END //

-- 3. FN_EstimarTiempoPostproduccion: Estima el tiempo necesario para completar la postproducción.
DELIMITER //
CREATE FUNCTION FN_EstimarTiempoPostproduccion(p_id_post INT)
RETURNS INT
DETERMINISTIC
BEGIN
    DECLARE v_inicio DATE;
    DECLARE v_fin DATE;
    DECLARE v_avance INT;
    DECLARE v_duracion_total INT;
    DECLARE v_tiempo_restante INT;

    SELECT Fecha_inicio, Fecha_fin, Avance_general
    INTO v_inicio, v_fin, v_avance
    FROM Postproduccion
    WHERE ID_post = p_id_post;

    SET v_duracion_total = DATEDIFF(v_fin, v_inicio);

    SET v_tiempo_restante = v_duracion_total * (100 - v_avance) / 100;

    RETURN v_tiempo_restante;
END //

DELIMITER ;

-- Ver dias restantes con un solo ID
SELECT FN_EstimarTiempoPostproduccion(1);

-- O con todas las tablas
SELECT 
    ID_post,
    FN_EstimarTiempoPostproduccion(ID_post) AS dias_restantes
FROM Postproduccion;


-- 4. FN_CalcularCostosDiarios: Calcula los costos diarios de producción según recursos asignados.
DELIMITER //
CREATE FUNCTION FN_CalcularCostosDiarios(p_id_plan INT)
RETURNS DECIMAL(15,2)
DETERMINISTIC
BEGIN
    DECLARE v_produccion INT;
    DECLARE costo_personal DECIMAL(15,2);
    DECLARE costo_equipo DECIMAL(15,2);
    DECLARE total DECIMAL(15,2);

    SELECT ID_produccion
    INTO v_produccion
    FROM Plan_Rodaje
    WHERE ID_plan = p_id_plan;

    SELECT IFNULL(SUM(p.Tarifa),0)
    INTO costo_personal
    FROM Produccion_Personal pp
    JOIN Personal p ON pp.ID_personal = p.ID_personal
    WHERE pp.ID_produccion = v_produccion;

    SELECT IFNULL(SUM(e.Valor),0)
    INTO costo_equipo
    FROM Plan_Rodaje_Equipo pe
    JOIN Equipo_Tecnico e ON pe.ID_equipo = e.ID_equipo
    WHERE pe.ID_plan = p_id_plan;

    SET total = costo_personal + costo_equipo;

    RETURN total;
END //

DELIMITER ;

-- Ver costos diarios con un solo ID
SELECT FN_CalcularCostosDiarios(5);

-- O con todas las tablas
SELECT 
    ID_plan,
    FN_CalcularCostosDiarios(ID_plan) AS costo_diario
FROM Plan_Rodaje;


-- 5. FN_VerificarPermisosLocalizacion: Verifica los permisos necesarios para una localización.
DELIMITER //
CREATE FUNCTION FN_VerificarPermisosLocalizacion(p_id_localizacion INT)
RETURNS VARCHAR(50)
DETERMINISTIC
BEGIN
    DECLARE v_permisos TEXT;
    DECLARE v_estado VARCHAR(50);

    SELECT Permisos, Estado
    INTO v_permisos, v_estado
    FROM Localizacion
    WHERE ID_localizacion = p_id_localizacion;

    IF v_permisos IS NOT NULL 
       AND v_permisos <> '' 
       AND v_estado = 'Disponible' THEN
        RETURN 'Aprobado';
    ELSE
        RETURN 'Sin permisos';
    END IF;
END //

DELIMITER ;

-- Ver permiso con un solo ID
SELECT FN_VerificarPermisosLocalizacion(9);

-- O con todas las tablas
SELECT 
    ID_localizacion,
    Nombre,
    FN_VerificarPermisosLocalizacion(ID_localizacion) AS estado_permisos
FROM Localizacion;
