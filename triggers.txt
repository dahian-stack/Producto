DELIMITER //

-- 2. TR_VerificarPresupuestoProduccion: Verifica que los gastos no excedan el presupuesto.
DELIMITER //

CREATE TRIGGER TR_VerificarPresupuestoProduccion
BEFORE UPDATE ON produccion
FOR EACH ROW 
BEGIN
    IF NEW.Presupuesto_gastado > NEW.Presupuesto_inicial THEN
       SIGNAL SQLSTATE '45000'
       SET MESSAGE_TEXT = 'Error: El gasto supera el presupuesto inicial';
    END IF;
END //

END //
